# Bayesian test
K <- 3 # Number of classes
delta = 0.0001

dir <- df |>
  group_by(RAWMATERIAL, ARTIFACTCLASS, STRAT_mod) |>
  count() |>
  pivot_wider(names_from = ARTIFACTCLASS, values_from = n, values_fill = 0) |>
  mutate(total = Core + Flake + RetouchedTool,
         
         Core_prop = Core/total,
         Flake_prop = Flake/total,
         RetouchedTool_prop = RetouchedTool/total,
         
         # Apply a negligible offset to account for 0 probabilities
         Core_propK = (Core + delta) / (total + K*delta),
         Flake_propK = (Flake + delta) / (total + K*delta),
         RetouchedTool_propK = (RetouchedTool + delta) / (total + K*delta),
         
         y = cbind(Core_propK, Flake_propK, RetouchedTool_propK)
         ) |>
  select(RAWMATERIAL, STRAT_mod,
         Core_prop, Core_propK,
         Flake_prop, Flake_propK,
         RetouchedTool_prop, RetouchedTool_propK,
         y)
